.equ SWI_Exit, 0x11 @ Assign a label to the terminate program code

@ Copy contents of first element of Vec1 to Vec2
	MOV R0, #0x29
	MOV R1, #0x10
ASC:
	SUB	R0,R0,R1
	CMP R0,#0x10
	ADD R3, R3, #1
	BGT	ASC
	MUL R3, R3, #0x20
	
	
	SWI	SWI_Exit	@ Terminate the program

.data	@ Define the data section

Vec1:	.word	1, 2, 3

Vec2:	.space	12
